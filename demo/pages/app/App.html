Simple Translation

../i18n/en-US.json

{{ '{' }} "greeting": {{ translateService.dictionaryManager.dictionaries[currentLocale].greeting | json }} {{ '}' }}

Pipe

'greeting' | translate
Example

{{ 'greeting' | translate }}

 

Directive

<p [translate]="'greeting'"></p>
Example

 

Controller

let message = translationService.translate('greeting');

Translation with Variables

../i18n/en-US.json

{{ '{' }} farewell": {{ translateService.dictionaryManager.dictionaries[currentLocale].farewell | json }} {{ '}' }}

Pipe

'farewell' | translate: {{ '{' }} name: 'Jane' {{ '}' }}
Example

{{ 'farewell' | translate: { name: 'Jane' } }}

 

Directive

<p [translate]="'farewell'" [dynamicValues]="{{ '{' }} name: 'Jane' {{ '}' }}"></p>
Example

 

Controller

let message = translationService.translate('farewell', {{ '{' }} name: 'Jane' {{ '}' }});
Example

 

Translation with Date Variables

../i18n/en-US.json

{{ '{' }} "today": {{ translateService.dictionaryManager.dictionaries[currentLocale].today | json }} {{ '}' }}
{{ '{' }} "formatToday": {{ translateService.dictionaryManager.dictionaries[currentLocale].formatToday | json }} {{ '}' }}

Can be used with syntax from momentjs.

Pipe

'today' | translate: {{ '{' }} today: {{ demoVariables.today }} {{ '}' }}
'formatToday' | translate: {{ '{' }} today: {{ demoVariables.today }} {{ '}' }}
Example

{{ 'today' | translate: { today: demoVariables.today } }}

{{ 'formatToday' | translate: { today: demoVariables.today } }}

 

Directive

<p [translate]="'today'" [dynamicValues]="{{ '{' }} today: {{ demoVariables.today }} {{ '}' }}"></p>
<p [translate]="'formatToday'" [dynamicValues]="{{ '{' }} today: {{ demoVariables.today }} {{ '}' }}"></p>
Example

 

Controller

let message = translationService.translate('today', {{ '{' }} today: {{ demoVariables.today }} {{ '}' }});
let message = translationService.translate('formatToday', {{ '{' }} today: {{ demoVariables.today }} {{ '}' }});
Example

Translation with Pluralization

../i18n/en-US.json

{{ '{' }} "messages": {{ translateService.dictionaryManager.dictionaries[currentLocale].messages | json }} {{ '}' }}

Pipe

'messages' | translate:0
'messages' | translate:1
'messages' | translate:20
'messages' | translate:50
Example

{{ 'messages' | translate:0 }}

{{ 'messages' | translate:1 }}

{{ 'messages' | translate:20 }}

{{ 'messages' | translate:50 }}

 

Directive

<p [translate]="'messages'" [dynamicValues]="0"></p>
<p [translate]="'messages'" [dynamicValues]="1"></p>
<p [translate]="'messages'" [dynamicValues]="20"></p>
<p [translate]="'messages'" [dynamicValues]="50"></p>
Example

 

Controller

let message = translationService.translate('messages', 0);
let message = translationService.translate('messages', 1);
let message = translationService.translate('messages', 20);
let message = translationService.translate('messages', 50);
Example

 

Translation with Pluralization With Dynamic Labels

../i18n/en-US.json

{{ '{' }} "candidate": {{ translateService.dictionaryManager.dictionaries[currentLocale].candidate | json }} {{ '}' }}

Pipe

'candidate' | translate:{{ '{' }} gender: 'female', quantity: 1, host: 'Jane', candidate: 'candidat' {{ '{' }}
'candidate' | translate:{{ '{' }} gender: 'male', quantity: 0, host: 'John', candidate: 'candidats' {{ '{' }}
'candidate' | translate:{{ '{' }} gender: 'other', quantity: 15, host: 'Les gens', candidate: 'candidats' {{ '{' }}
Example

{{ 'candidate' | translate:{ gender: 'female', quantity: 1, host: 'Jane', candidate: 'candidats' } }}

{{ 'candidate' | translate:{ gender: 'male', quantity: 0, host: 'John', candidate: 'candidats' } }}

{{ 'candidate' | translate:{ gender: 'other', quantity: 0, host: 'Les gens', candidate: 'candidats' } }}

 

Directive

<p [translate]="'candidate'" [dynamicValues]="{{ '{' }} gender: 'female', quantity: 1, host: 'Jane', candidate: 'candidat' {{ '{' }}"></p>
<p [translate]="'candidate'" [dynamicValues]="{{ '{' }} gender: 'male', quantity: 0, host: 'John', candidate: 'candidats' {{ '{' }}"></p>
<p [translate]="'candidate'" [dynamicValues]="{{ '{' }} gender: 'other', quantity: 15, host: 'Les gens', candidate: 'candidats' {{ '{' }}"></p>
Example

 

Controller

let message = translationService.translate('candidate', {{ '{' }} gender: 'female', quantity: 1, host: 'Jane', candidate: 'candidat' {{ '{' }});
let message = translationService.translate('candidate', {{ '{' }} gender: 'male', quantity: 0, host: 'John', candidate: 'candidats' {{ '{' }});
let message = translationService.translate('candidate', {{ '{' }} gender: 'other', quantity: 15, host: 'Les gens', candidate: 'candidats' {{ '{' }});
